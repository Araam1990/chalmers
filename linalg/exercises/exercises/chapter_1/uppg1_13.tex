\begin{math}
    v=\begin{pmatrix}
        1\\
        3\\
        2
    \end{pmatrix},
    \hspace{6px}
    w=\begin{pmatrix}
        3\\
        2\\
        -1
    \end{pmatrix}
    \\
    ||v||
    =\sqrt{1^2+3^2+2^2}
    =\sqrt{1+9+4}
    =\sqrt{14}
    \\
    ||w||
    =\sqrt{3^2+2^2+(-1)^2}
    =\sqrt{9+4+1}
    =\sqrt{14}
    \\
    v\cdot w
    =1*3+3*2+2*(-1)
    =3+6-2
    =7
    \\
    \cos(\theta)
    =\frac{v\cdot w}{||v||*||w||}
    =\frac{7}{\sqrt{14}*\sqrt{14}}
    =\frac{7}{14}
    =\frac{1}{2}
    \\
    \theta
    =\cos^{-1}(\frac{1}{2})
    =60^{\circ}
\end{math}
